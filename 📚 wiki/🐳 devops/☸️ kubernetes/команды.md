```bash
kubectl apply -f <yaml file>   # создание ресурсов из файла
kubectl get                    # вывод списка всех ресурсов
kubectl get deploymets         # получение всех развертываний
kubectl get rs                 # получение всех ReplicaSet
kubectl get pods               # показ всех под
kubectl describe <resources>   # получение информации о ресурсах
kubectl logs <container>       # получение логов контейнера в поде
kubectl exec <id> -- <command> # выполнения команды в контейнере
kubectl delete <resource> <id> # удаление ресурса
kubectl get <resource> --watch # просмотр в реальном времени
kubectl scale <resource> --replicas=<number>

kubectl set image <resource> <name> 
kubectl rollout status deployments kubernetes-bootcamp
kubectl rollout undo deployments kubernetes-bootcamp

kubectl describe services kubernetes-bootcamp

```
